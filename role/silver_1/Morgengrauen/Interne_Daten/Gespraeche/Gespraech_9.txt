#Vor dir steht ein junger Elf, der neugierig die [Waren] mustert.

RACE 3 IstElf

§anfang
  INPUT name beruf alter ende waren
  #Dazu kann ich nichts sagen.
  GOTO anfang


§beruf
  #Beruf? Was ist das? Etwas zu Essen?
  GOTO anfang

§alter
  #Weiß ich nicht... Ich habe die Sommer nicht gezählt.
  GOTO anfang

§ende
  #Bis bald!
  END

§name
  #Asfelion - und du?
  GOTO anfang

§waren
  #Oh, hier gibt es einige interessante Sachen - zumindest wenn man weiß,
  #was man mit ihnen tun kann...
  #Ich weiß es... Kannst du eigentlich zaubern?
  YN kannzaubern

  #Schade - ist manchmal sehr hilfreich...
  #Aber Ich kann es! Wenn du willst, begleite ich dich. Ich kann dir
  #sicher oft hilfreich sein!
  #Was sagst du dazu? Darf ich dich begleiten?
  YN darfbegleiten
  #Dann eben nicht.
  GOTO anfang

§kannzaubern
  #Fein! Kannst du mir was beibringen?
  #Oder besser noch - darf ich dich begleiten?
  YN darfbegleiten
  #Dann eben nicht.
  GOTO anfang


§darfbegleiten
  FOLLOW 5 fehler
  SET 55
  END

§fehler
  #Dann eben nicht.
  GOTO anfang



§IstElf
  #Es ist schön, jemanden aus meinem Volk zu treffen.
  #Mein Name ist Asfelion
