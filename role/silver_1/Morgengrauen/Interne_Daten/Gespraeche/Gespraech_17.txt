§legenden
  IFSET 80 auftragerhalten
    #Nun, zur Zeit erforsche ich die Sage des Sonnenbaums... Es heißt, dieser
    #Baum steht inmitten des Sumpfes! Natürlich kann ich mich nicht selbst
    #davon überzeugen, daß es diesen Baum wirklich gibt - aber für Berichte
    #über diesen Baum habe ich ein offenes Ohr.
    SET 80

    §auftragerhalten

      IFSET 78 baumgefunden
	#Solltet ihr etwas über den Sonnenbaum hören, dann berichtet es mir bitte.
	GOTO anfang

§baumgefunden
  IFSET 79 belohnungerhalten
    #Wirklich?! Ihr habt diesen Baum gesehen?! Erzählt mir mehr!
    #(Es folgt nun der Bericht...)
    #Habt vielen Dank!
    SET 79
    EP 20
    MONEY 100
    GOTO anfang

  §belohnungerhalten
    #Schön euch nochmals zu sehen...
    GOTO anfang
